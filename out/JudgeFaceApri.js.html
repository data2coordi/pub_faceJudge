/**
 * FaceApiClassクラスはFaceAPIに関連する機能を提供します。
 */
class FaceApiClass {
    /**
     * モデルを作成します。
     * @returns {Promise} Promiseオブジェクト
     */
    static create_model() {
        return Promise.all([
            faceapi.nets.tinyFaceDetector.loadFromUri('models'),
            faceapi.nets.faceLandmark68TinyNet.loadFromUri('models'),
            faceapi.nets.faceLandmark68Net.load("models"),
            faceapi.nets.ssdMobilenetv1.loadFromUri('models')
        ]);
    }

    /**
     * メディアからCanvasを作成します。
     * @param {HTMLImageElement} start_img - 開始画像のHTMLImageElement
     * @returns {HTMLCanvasElement} 作成されたCanvas要素
     */
    static createCanvasFromMedia(start_img) {
        return faceapi.createCanvasFromMedia(start_img);
    }

    /**
     * 与えられた画像からすべての顔を検出し、顔のランドマークを含むPromiseを返します。
     * @param {HTMLImageElement} start_img - 開始画像のHTMLImageElement
     * @returns {Promise} Promiseオブジェクト
     */
    static detectAllFaces(start_img) {
        return faceapi.detectAllFaces(start_img).withFaceLandmarks();
    }
}
